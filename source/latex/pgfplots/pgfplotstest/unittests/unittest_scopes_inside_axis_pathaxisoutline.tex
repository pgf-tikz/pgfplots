		\begin{axis}[
			#1,
			clip=false,
		]
		\addplot+[samples=3] {x^2};
		\begin{scope}
			\clip \pgfextra{\pgfplotspathaxisoutline};

			\draw (axis cs:-20,15) -- (axis cs:20,15);

			\draw (axis cs:-20,20) -- (axis cs:20,20);
		\end{scope}
		\addplot+[samples=3] {x};
		\end{axis}
	\end{tikzpicture}
}

	clip marker paths,
}
