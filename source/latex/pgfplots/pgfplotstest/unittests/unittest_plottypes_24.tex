\begin{document}
\begin{tikzpicture}
	\begin{axis}[
		domain=0:1,
		xmax=1,
		ymax=1,
	]
		\addplot3[surf] {x*y};
		\addplot3[blue,/pgfplots/quiver,
			quiver/u=0.1*y,
			quiver/v=0.1*x,
			quiver/w value=0,
			-stealth,samples=10] {1};
	\end{axis}
\end{tikzpicture}
